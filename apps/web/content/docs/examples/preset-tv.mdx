# TV / EPG preset

Configure the scheduler as an Electronic Programme Guide — channels as rows, programmes as blocks spanning horizontal time slots.

[→ Try it live](/demo)

```tsx
import { Scheduler, createTvConfig } from '@sushill/shadcn-scheduler'

const channels = [
  { id: 'bbc1',  name: 'BBC One',   kind: 'category' as const, colorIdx: 0 },
  { id: 'bbc2',  name: 'BBC Two',   kind: 'category' as const, colorIdx: 1 },
  { id: 'itv',   name: 'ITV',       kind: 'category' as const, colorIdx: 2 },
  { id: 'ch4',   name: 'Channel 4', kind: 'category' as const, colorIdx: 3 },
]

const programmes = [
  { id: 'p1', categoryId: 'bbc1', employeeId: 'bbc1', employee: 'News at Six',
    date: '2026-03-19', startH: 18, endH: 18.5, status: 'published' as const },
  { id: 'p2', categoryId: 'bbc1', employeeId: 'bbc1', employee: 'EastEnders',
    date: '2026-03-19', startH: 19.5, endH: 20, status: 'published' as const },
]

const config = createTvConfig()

<Scheduler
  categories={channels}
  employees={channels}   // In TV mode, categories = employees (no sub-resources)
  shifts={programmes}
  onShiftsChange={setProgrammes}
  config={config}
  initialView="timeline"
/>
```
