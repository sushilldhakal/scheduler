# Gantt preset

Visualise project timelines with tasks as blocks spanning multiple days. Teams are categories, team members are employees, tasks are blocks.

[→ Try it live](/demo)

```tsx
import { Scheduler, createGanttConfig } from '@sushill/shadcn-scheduler'

const teams = [
  { id: 'frontend', name: 'Frontend',  kind: 'category' as const, colorIdx: 0 },
  { id: 'backend',  name: 'Backend',   kind: 'category' as const, colorIdx: 2 },
  { id: 'design',   name: 'Design',    kind: 'category' as const, colorIdx: 4 },
]

const tasks = [
  { id: 't1', categoryId: 'frontend', employeeId: 'dev1', employee: 'Alice',
    date: '2026-03-19', startH: 0, endH: 8, status: 'published' as const },
]

const config = createGanttConfig()
// Labels: Project, Team member, Task
// Day-wide blocks (startH=0, endH=24 for full-day tasks)
```
