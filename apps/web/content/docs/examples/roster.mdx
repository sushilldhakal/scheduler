# Workforce roster

A realistic workforce roster with multiple departments, employees, and pre-populated shifts for the current week.

[→ Try it live](/demo)

```tsx
'use client'
import { useState } from 'react'
import { Scheduler, createSchedulerConfig, type Block, type Resource } from '@sushill/shadcn-scheduler'

const categories: Resource[] = [
  { id: 'front-desk', name: 'Front Desk', kind: 'category', colorIdx: 0 },
  { id: 'kitchen',    name: 'Kitchen',    kind: 'category', colorIdx: 2 },
  { id: 'management', name: 'Management', kind: 'category', colorIdx: 4 },
]

const employees: Resource[] = [
  { id: 'e1', name: 'Alice',   kind: 'employee', categoryId: 'front-desk', colorIdx: 0 },
  { id: 'e2', name: 'Bob',     kind: 'employee', categoryId: 'front-desk', colorIdx: 0 },
  { id: 'e3', name: 'Carlos',  kind: 'employee', categoryId: 'kitchen',    colorIdx: 2 },
  { id: 'e4', name: 'Diana',   kind: 'employee', categoryId: 'kitchen',    colorIdx: 2 },
  { id: 'e5', name: 'Edward',  kind: 'employee', categoryId: 'management', colorIdx: 4 },
]

const today = new Date().toISOString().split('T')[0]

const initialShifts: Block[] = [
  { id: 's1', categoryId: 'front-desk', employeeId: 'e1', employee: 'Alice',   date: today, startH: 8,  endH: 16, status: 'published' },
  { id: 's2', categoryId: 'front-desk', employeeId: 'e2', employee: 'Bob',     date: today, startH: 14, endH: 22, status: 'published' },
  { id: 's3', categoryId: 'kitchen',    employeeId: 'e3', employee: 'Carlos',  date: today, startH: 9,  endH: 17, status: 'draft'     },
  { id: 's4', categoryId: 'kitchen',    employeeId: 'e4', employee: 'Diana',   date: today, startH: 17, endH: 23, status: 'published' },
  { id: 's5', categoryId: 'management', employeeId: 'e5', employee: 'Edward',  date: today, startH: 9,  endH: 18, status: 'published' },
]

const config = createSchedulerConfig({ preset: 'roster', {
  initialScrollToNow: true,
  workingHours: {
    0: null,
    1: { from: 7, to: 23 },
    2: { from: 7, to: 23 },
    3: { from: 7, to: 23 },
    4: { from: 7, to: 23 },
    5: { from: 7, to: 23 },
    6: { from: 8, to: 20 },
  },
} })

export default function RosterPage() {
  const [shifts, setShifts] = useState(initialShifts)
  return (
    <div className="h-[600px]">
      <Scheduler
        categories={categories}
        employees={employees}
        shifts={shifts}
        onShiftsChange={setShifts}
        initialView="week"
        config={config}
      />
    </div>
  )
}
```
